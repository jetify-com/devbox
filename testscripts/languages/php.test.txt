-- devbox.json --
{
  "packages": [

  ]

-- index.php --
<?php

// Check that the extension is loaded.
if (!extension_loaded('ds')) {
    echo("ds extension is not enabled");
    exit(0);
}

$seq = new \Ds\Seq(["hello", "world"]);

echo("Original sequence elements\n");
foreach ($seq as $idx => $elem) {
  echo("idx: $idx and elem: $elem\n");
}
echo("done\n");
